\section{What is it?}
\frame {
    \frametitle{What is it?}
    \begin{itemize}
    \item A process runtime engine (E.g. Business processes)
    \item Based on UML2 activity model specification (Metamodel)
    \item Which interprets rather than compiles models
    \item Allows runtime-modification of the model
    \item Programmed in Python using Zope
    \end{itemize}
    \begin{itemize}
    \item $\rightarrow$ Activity Model Runtime Engine for Python
    \end{itemize}
}

    \addvspace{5mm}
    Action node delegates execution to external implementation\\
    \addvspace{5mm}

        \texttt{start(runtime\_data=None)},
        \texttt{next}
    }
